const NOMBRE_HOJA_CAPTURAS = "CAPTURAS"; const NOMBRE_HOJA_USUARIOS = "USUARIOS"; const NOMBRE_CARPETA_EVIDENCIAS = "EVIDENCIAS_SCANNER"; function doGet(e) { const accion = e.parameter.accion; if (accion === "buscar_pandape") { const pandape = String(e.parameter.pandape || "").trim().toUpperCase(); const callback = e.parameter.callback; const respuesta = buscarPandape(pandape); const json = JSON.stringify(respuesta); if (callback) { return ContentService .createTextOutput(callback + "(" + json + ")") .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput("OK"); } function buscarPandape(pandape) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const hoja = ss.getSheetByName(NOMBRE_HOJA_USUARIOS); if (!hoja) { return { encontrado: false, mensaje: "No existe la hoja USUARIOS" }; } const datos = hoja.getDataRange().getValues(); if (datos.length < 2) { return { encontrado: false, mensaje: "La hoja USUARIOS no tiene datos" }; } const encabezados = datos[0].map(e => normalizarTexto(e)); let idxPandape = encabezados.findIndex(e => e === "PANDAPE"); let idxNombre = encabezados.findIndex(e => e === "NOMBRE_COMPLETO" || e === "NOMBRE COMPLETO" || e.includes("NOMBRE")); if (idxPandape === -1) idxPandape = 0; if (idxNombre === -1) idxNombre = 1; for (let i = 1; i < datos.length; i++) { const pandapeHoja = String(datos[i][idxPandape] || "").trim().toUpperCase(); const nombreCompleto = String(datos[i][idxNombre] || "").trim(); if (pandapeHoja === pandape) { return { encontrado: true, pandape: pandapeHoja, nombre_completo: nombreCompleto }; } } return { encontrado: false, mensaje: "Pandape no encontrado" }; } function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let hoja = ss.getSheetByName(NOMBRE_HOJA_CAPTURAS); if (!hoja) { hoja = ss.insertSheet(NOMBRE_HOJA_CAPTURAS); } prepararHojaCapturas(hoja); const p = e.parameter; const fechaArchivo = Utilities.formatDate( new Date(), "America/Mexico_City", "yyyyMMdd_HHmmss" ); const folio = limpiarNombreArchivo(p.folio || "SIN_FOLIO"); const pandape = limpiarNombreArchivo(p.pandape || "SIN_PANDAPE"); const urlQR = guardarImagenBase64( p.evidencia_qr_base64 || "", "QR_" + folio + "_" + pandape + "_" + fechaArchivo + ".jpg" ); const urlCompra = guardarImagenBase64( p.evidencia_compra_base64 || "", "COMPRA_" + folio + "_" + pandape + "_" + fechaArchivo + ".jpg" ); hoja.appendRow([ p.fecha_captura || new Date(), p.pandape || "", p.nombre_completo || "", p.quien_registro || "", p.codigo_barras || "", p.ultimos4 || "", p.folio || "", p.determinante || "", p.tienda || "", p.fecha_venta || "", p.comentarios || "", urlQR, urlCompra ]); return ContentService.createTextOutput("OK"); } function prepararHojaCapturas(hoja) { const encabezados = [ "FECHA_CAPTURA", "PANDAPE", "NOMBRE_COMPLETO", "QUIEN_REGISTRO", "CODIGO_BARRAS", "ULTIMOS_4_TARJETA", "FOLIO", "DETERMINANTE", "TIENDA", "FECHA_VENTA", "COMENTARIOS", "LINK_EVIDENCIA_QR", "LINK_EVIDENCIA_COMPRA_MENSUAL" ]; const primeraCelda = hoja.getRange(1, 1).getValue(); if (!primeraCelda) { hoja.getRange(1, 1, 1, encabezados.length).setValues([encabezados]); hoja.setFrozenRows(1); return; } const filaActual = hoja.getRange(1, 1, 1, encabezados.length).getValues()[0]; const necesitaActualizar = encabezados.some((h, i) => filaActual[i] !== h); if (necesitaActualizar) { hoja.getRange(1, 1, 1, encabezados.length).setValues([encabezados]); hoja.setFrozenRows(1); } } function guardarImagenBase64(base64, nombreArchivo) { if (!base64) return ""; const partes = base64.split(","); if (partes.length < 2) return ""; const metadata = partes[0]; const contenido = partes[1]; const mimeMatch = metadata.match(/data:(.*);base64/); const mimeType = mimeMatch ? mimeMatch[1] : "image/jpeg"; const bytes = Utilities.base64Decode(contenido); const blob = Utilities.newBlob(bytes, mimeType, nombreArchivo); const carpeta = obtenerCarpetaEvidencias(); const archivo = carpeta.createFile(blob); try { archivo.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); } catch (error) { Logger.log("No se pudo cambiar permisos del archivo: " + error); } return archivo.getUrl(); } function obtenerCarpetaEvidencias() { const carpetas = DriveApp.getFoldersByName(NOMBRE_CARPETA_EVIDENCIAS); if (carpetas.hasNext()) { return carpetas.next(); } return DriveApp.createFolder(NOMBRE_CARPETA_EVIDENCIAS); } function limpiarNombreArchivo(texto) { return String(texto || "") .trim() .replace(/[^\w\-]/g, "_") .substring(0, 80); } function normalizarTexto(texto) { return String(texto || "") .trim() .toUpperCase() .normalize("NFD") .replace(/[\u0300-\u036f]/g, ""); }
